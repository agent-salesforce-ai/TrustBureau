function renderSavedOffers() { if (!elements.savedOffersSection || !elements.savedOffersContainer) return; const used = savedOffers.length; if (used === 0) { elements.savedOffersSection.classList.add('hidden'); elements.savedOffersContainer.innerHTML = ''; return; } elements.savedOffersSection.classList.remove('hidden'); const cardsHtml = savedOffers.map((offer, index) => { const labelIndex = index + 2; // Offer 2 and 3 return `
Offer ${labelIndex} ${offer.paymentLabel}
You Get ${formatCurrency(offer.amount)}
Factor ${offer.factor.toFixed(2)}x
Term ${offer.termMonths} mo
Payback ${formatCurrency(offer.totalPayback)}
${offer.paymentLabel} ${formatCurrency(offer.payment)}
`; }).join(''); elements.savedOffersContainer.innerHTML = cardsHtml; // Each additional-offer slider is functional only for its own card on this link. // It does NOT change savedOffers, so rep-side approvals (and future links) stay intact. const sliders = elements.savedOffersContainer.querySelectorAll('.saved-offer-slider'); sliders.forEach((slider) => { slider.addEventListener('input', () => { const idxAttr = slider.getAttribute('data-offer-index'); const idx = idxAttr != null ? parseInt(idxAttr, 10) : -1; if (Number.isNaN(idx) || idx < 0 || idx >= savedOffers.length) return; let newAmount = parseFloat(slider.value); if (isNaN(newAmount)) return; const cap = parseFloat(slider.max) || savedOffers[idx].amount || 0; if (newAmount > cap) newAmount = cap; if (newAmount < 5000) newAmount = 5000; slider.value = newAmount; const baseFactor = parseFloat(slider.dataset.factor) || savedOffers[idx].factor || 1.1; const termMonths = parseInt(slider.dataset.term, 10) || savedOffers[idx].termMonths || 3; const freq = slider.dataset.frequency || savedOffers[idx].frequency || 'daily'; const updated = computeOffer(newAmount, baseFactor, termMonths, freq); const card = slider.closest('[data-offer-index]'); if (!card) return; const amtEl = card.querySelector('[data-role="amount"]'); const factorEl = card.querySelector('[data-role="factor"]'); const termEl = card.querySelector('[data-role="term"]'); const paybackEl = card.querySelector('[data-role="payback"]'); const payLabelEl = card.querySelector('[data-role="payment-label"]'); const paymentEl = card.querySelector('[data-role="payment"]'); if (amtEl) amtEl.textContent = formatCurrency(updated.amount); if (factorEl) factorEl.textContent = `${updated.factor.toFixed(2)}x`; if (termEl) termEl.textContent = `${updated.termMonths} mo`; if (paybackEl) paybackEl.textContent = formatCurrency(updated.totalPayback); if (payLabelEl) payLabelEl.textContent = updated.paymentLabel; if (paymentEl) paymentEl.textContent = formatCurrency(updated.payment); }); }); }
